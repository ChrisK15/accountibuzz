```ts
function cutoffLabelFor(timezone: string): string {
  // Format "9:00 PM" — the group's local cutoff is midnight; we format midnight in the group's tz.
  // For UI-SPEC §"Still to post" line 355, the cutoff is the group's deadline.
  // The existing screen at line 87 calls `cutoffStateFor` (P3 helper) — reuse if signature matches;
  // otherwise compose via Intl.DateTimeFormat below.
  // Conservative implementation: format midnight tomorrow in the group's tz as 12:00 AM, then
  // back off one minute to display "11:59 PM" — but UI-SPEC just says "9:00 PM cutoff" as the
  // example, implying the group cutoff is configurable. For MVP, return the group's local
  // midnight formatted as time-of-day:
  const tomorrow = new Date();
  tomorrow.setDate(tomorrow.getDate() + 1);
  tomorrow.setHours(0, 0, 0, 0);
  return new Intl.DateTimeFormat('en-US', {
```
