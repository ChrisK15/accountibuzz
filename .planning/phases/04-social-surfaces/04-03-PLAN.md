  userId: string | undefined,

            event: '*',

            table: 'submissions',
            filter: `user_id=eq.${userId}`,
          } as never,
          (payload) => {
            // ... narrowing + setQueryData ...
          },
        )
        .subscribe();
      return () => { supabase.removeChannel(channel); };
    }, [userId, qc, getGroupTzs]),
  );
}
