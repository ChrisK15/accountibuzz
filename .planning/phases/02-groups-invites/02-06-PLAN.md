    ```tsx
    import { useState } from 'react';
    import { KeyboardAvoidingView, Platform, View, Text, Pressable } from 'react-native';
    import { useRouter, Stack } from 'expo-router';
    import { useForm, Controller } from 'react-hook-form';
    import { zodResolver } from '@hookform/resolvers/zod';
    import { Feather } from '@expo/vector-icons';
    import { joinCodeSchema, type JoinCodeInput } from '../../../src/features/groups/schemas';
    import { normalizeInviteCode, formatInviteCode } from '../../../src/features/groups/formatInviteCode';
    import { useRedeemInvite } from '../../../src/features/groups/useRedeemInvite';
    import { useTheme } from '../../../src/theme/useTheme';
    import {
      ScreenContainer, ScreenHeader, TextInput, FormError, PrimaryButton,
    } from '../../../src/components';

    export default function JoinScreen() {
      const t = useTheme();
      const router = useRouter();
      const redeem = useRedeemInvite();
      const [submitError, setSubmitError] = useState<string | null>(null);

      const { control, handleSubmit, watch, setValue, formState: { errors, isValid, isSubmitting } } =
        useForm<JoinCodeInput>({
          resolver: zodResolver(joinCodeSchema),
          mode: 'onBlur',
          defaultValues: { code: '' },
        });

      const displayValue = (() => {
        const raw = watch('code') ?? '';
        return raw.length > 4 ? `${raw.slice(0, 4)}-${raw.slice(4)}` : raw;
      })();

      const onChangeText = (input: string) => {
        // Normalize on every keystroke: strip + uppercase + slice to 8.
        const raw = normalizeInviteCode(input);
        setValue('code', raw, { shouldValidate: true });
      };

      const onSubmit = handleSubmit(async (values) => {
        setSubmitError(null);
        try {
          const groupId = await redeem.mutateAsync(values.code);
          router.replace(`/groups/${groupId}`);
        } catch (err: unknown) {
          const msg = err instanceof Error ? err.message : '';
          switch (msg) {
            case 'group_full':
              setSubmitError("This group's already at 10 members. Ask the admin to make room or start your own.");
              break;
            case 'invite_expired':
              setSubmitError('This invite expired. Ask the admin for a fresh code.');
              break;
            case 'invite_already_used':
              setSubmitError("This code's already been used. Ask the admin for a new one.");
              break;
            case 'already_member':
              setSubmitError("You're already in this group. Head on over.");
              break;
            case 'invite_not_found':
              setSubmitError("We don't know that code. Double-check it with whoever invited you.");
              break;
            default:
              setSubmitError('Something went sideways. Check your connection and try again.');
          }
        }
      });

      return (
        <KeyboardAvoidingView behavior={Platform.OS === 'ios' ? 'padding' : undefined} style={{ flex: 1 }}>
          <Stack.Screen options={{ headerShown: false }} />
          <ScreenContainer>
            <View style={{ flexDirection: 'row', alignItems: 'center', padding: t.spacing.xl, gap: t.spacing.md }}>
              <Pressable onPress={() => router.back()} accessibilityLabel="Back" hitSlop={8}>
                <Feather name="chevron-left" size={24} color={t.colors.textStrong} />
              </Pressable>
              <Text style={t.fonts.heading2}>Got a code?</Text>
            </View>

            <ScreenHeader title="Got a code?" subtitle="Enter it below to join your friends." />

            <View style={{ padding: t.spacing.xl, gap: t.spacing.lg }}>
              <Controller
                control={control}
                name="code"
                render={() => (
                  <TextInput
                    label="Invite code"
                    placeholder="ABCD-EF12"
                    value={displayValue}
                    onChangeText={onChangeText}
                    autoCapitalize="characters"
                    autoCorrect={false}
                    maxLength={9} // 8 chars + 1 optional dash
                    keyboardType="default"
                    error={errors.code?.message}
                    helper="8 letters and numbers. Dashes optional."
                    inputStyle={{
                      fontFamily: 'Manrope_700Bold',
                      fontSize: 20,
                      letterSpacing: 2,
                      fontVariant: ['tabular-nums'],
                    }}
                  />
                )}
              />

              {submitError && <FormError>{submitError}</FormError>}

              <PrimaryButton
                label="Join group"
                onPress={onSubmit}
                loading={isSubmitting || redeem.isPending}
                disabled={!isValid}
              />
            </View>
          </ScreenContainer>
        </KeyboardAvoidingView>
      );
    }
    ```

    ```tsx
    import { useLocalSearchParams, useRouter, Stack } from 'expo-router';
    import { View, Text, Pressable } from 'react-native';
    import * as SecureStore from 'expo-secure-store';
    import { useInvitePreview } from '../../src/features/groups/useInvitePreview';
    import { useRedeemInvite } from '../../src/features/groups/useRedeemInvite';
    import { useSession } from '../../src/features/auth/AuthProvider';
    import { PENDING_INVITE_KEY } from '../../src/features/groups/usePendingInviteReplay';
    import { useTheme } from '../../src/theme/useTheme';
    import {
      ScreenContainer, ScreenHeader, PrimaryButton, GhostButton, Avatar,
    } from '../../src/components';
    import { useState } from 'react';

    export default function InviteLandingScreen() {
      const t = useTheme();
      const router = useRouter();
      const { code: rawCode } = useLocalSearchParams<{ code: string }>();
      const code = (rawCode ?? '').toUpperCase();
      const { session, loading: sessionLoading } = useSession();
      const { data: preview, isPending: previewPending, error: previewError } =
        useInvitePreview(code);
      const redeem = useRedeemInvite();
      const [joinError, setJoinError] = useState<string | null>(null);

      if (sessionLoading || previewPending) {
        return (
          <ScreenContainer>
            <Stack.Screen options={{ headerShown: false }} />
            <SkeletonPreviewCard />
          </ScreenContainer>
        );
      }

      if (previewError) {
        return (
          <ScreenContainer>
            <Stack.Screen options={{ headerShown: false }} />
            <ScreenHeader
              title="Invite not found"
              subtitle="We don't know that code. Double-check it with whoever invited you."
            />
            <GhostButton label="Back to groups" onPress={() => router.replace('/')} />
          </ScreenContainer>
        );
      }

      if (!session || !preview) {
        // Unauthenticated branch
        return (
          <ScreenContainer>
            <Stack.Screen options={{ headerShown: false }} />
            <View style={{ padding: t.spacing.xl, gap: t.spacing.lg, alignItems: 'center' }}>
              <Text style={t.fonts.display}>{`${preview?.admin_display_name ?? 'Someone'} invited you`}</Text>
              <Text style={t.fonts.body}>
                {`Join ${preview?.group_name ?? 'the group'} — ${preview?.member_count ?? 0} already in.`}
              </Text>
              <PrimaryButton
                label="Sign in to join"
                onPress={async () => {
                  await SecureStore.setItemAsync(PENDING_INVITE_KEY, code);
                  router.replace('/(auth)/login');
                }}
              />
              <Pressable
                onPress={async () => {
                  await SecureStore.setItemAsync(PENDING_INVITE_KEY, code);
                  router.replace('/(auth)/signup');
                }}
              >
                <Text style={[t.fonts.body, { color: t.colors.accent, fontWeight: '700' }]}>
                  Create an account to join
                </Text>
              </Pressable>
            </View>
          </ScreenContainer>
        );
      }

      // Authenticated branch — ready to redeem
      const onJoin = async () => {
        setJoinError(null);
        try {
          const groupId = await redeem.mutateAsync(code);
          // useRedeemInvite.onSuccess already clears PENDING_INVITE_KEY
          router.replace(`/groups/${groupId}`);
        } catch (err: unknown) {
          const msg = err instanceof Error ? err.message : '';
          switch (msg) {
            case 'group_full':
              setJoinError("This group's already at 10 members. Ask the admin to make room or start your own.");
              break;
            case 'invite_expired':
              setJoinError('This invite expired. Ask the admin for a fresh code.');
              break;
            case 'invite_already_used':
              setJoinError("This code's already been used. Ask the admin for a new one.");
              break;
            case 'already_member':
              setJoinError("You're already in this group. Head on over.");
              break;
            case 'invite_not_found':
              setJoinError("We don't know that code. Double-check it with whoever invited you.");
              break;
            default:
              setJoinError('Something went sideways. Check your connection and try again.');
          }
        }
      };

      return (
        <ScreenContainer>
          <Stack.Screen options={{ headerShown: false }} />
          <View style={{ padding: t.spacing.xl, gap: t.spacing.lg, alignItems: 'center' }}>
            <Text style={t.fonts.display}>Ready to join?</Text>
            <Text style={t.fonts.body}>{`You're about to join ${preview.group_name}.`}</Text>
            <PrimaryButton label="Join group" onPress={onJoin} loading={redeem.isPending} />
            {joinError && <Text style={[t.fonts.body, { color: t.colors.destructive }]}>{joinError}</Text>}
          </View>
        </ScreenContainer>
      );
    }
    ```
