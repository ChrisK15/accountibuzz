**Lifecycle + channel-subscribe pattern** (analog lines 42-117 — copy verbatim, swap channel name + filter + narrowing):
```ts
export function useGroupFeedRealtime(
  groupId: string | undefined,
  groupTimezone: string | undefined,
): void {
  const qc = useQueryClient();
  useFocusEffect(
    useCallback(() => {
      if (!groupId || !groupTimezone) return;

      const channel = supabase
        .channel(`group-feed:${groupId}`)
        .on(
          'postgres_changes' as never,
          {
            event: '*',
            schema: 'public',
            table: 'submissions',
            filter: `group_id=eq.${groupId}`,
          } as never,
          (payload: { new?: unknown; old?: unknown }) => {
            // ... narrowing + patch ...
          },
        )
        .subscribe();

      return () => {
        supabase.removeChannel(channel);
      };
    }, [groupId, groupTimezone, qc]),
  );
}
```

**Narrowing pattern** (adapted from analog lines 67-86 — change "approved" → "pending"; keep `oldRow.status` reads — they are reliable because P4 migration `20260508233129` set `replica identity full` on `submissions`, see analog header comment lines 8-14):
```ts
const newRow = payload.new as
  | { status: 'pending' | 'approved' | 'rejected'; id: string }
  | undefined;
const oldRow = payload.old as
  | { status?: 'pending' | 'approved' | 'rejected' }
  | undefined;

const insertedPending = !oldRow && newRow?.status === 'pending';
const flippedFromPending =
  oldRow?.status === 'pending' && newRow?.status !== 'pending';
const flippedToPending =
  oldRow?.status && oldRow.status !== 'pending' &&
  newRow?.status === 'pending';
if (!insertedPending && !flippedFromPending && !flippedToPending) {
  return;
}
```
